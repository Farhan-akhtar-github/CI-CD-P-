name: CI-CD Pipeline

on:               # trigger
  push:
    branches:
      - main

jobs:             # A job is a group of steps that run together on the same runner (machine).
  ci:
    name: CI - Test, Build & Push Docker Image
    runs-on: ubuntu-latest

    steps:
      - name: Checkout code
        uses: actions/checkout@v4        # actions/checkout@v4 downloads (checks out) your repository code into the GitHub Actions runner so the workflow can use it

      - name: Set up Python
        uses: actions/setup-python@v5    # setup python
        with:                            # "with" is used to pass input parameters
          python-version: '3.11'

      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip    # pip is Python’s package manager

                                                 #python -m pip install --upgrade pip upgrades pip using the same Python interpreter, ensuring the package manager is up to date for that            						  Python version.
          pip install -r requirements.txt

      - name: Run unit tests
        run: |
          pytest tests/

      - name: Login to Docker Hub
        run: |
          echo "${{ secrets.DOCKER_PASSWORD }}" | docker login \       # echo - Prints the password to stdout (standard output)
 
                                                                       # echo "${{ secrets.DOCKER_PASSWORD }}" → sends output

                                                                       # Pipe (|) → redirects output to input

                                                                       # | (pipe) 

                                                                            1.Pipe takes stdout of echo

                                                                            2.Sends it to stdin of docker login
 
                                                                       # The \ is used for line continuation.Shell joins lines internally and runs:docker login -u username --password-stdin


          -u "${{ secrets.DOCKER_USERNAME }}" --password-stdin         # -u :-  -u means username

                                                                       # We don’t use echo for the username because Docker only supports passing the password via stdin. The                     						                       username is not sensitive and is safely passed as a command-line flag using -u.
                                                                
                                                                       # --password-stdin is used to securely pass the password to Docker via standard input instead of exposing it in the                       									 command line, logs, or shell history.
      - name: Build Docker image
        run: |
          docker build -t ${{ secrets.DOCKER_USERNAME }}/ci-cd-clean-app:latest .  # -t It assigns a name and version to the Docker image at build time.

                                                                                   # ci-cd-clean-app:- Application image name,  latest :- Tag for newest build.


      - name: Push Docker image
        run: |
          docker push ${{ secrets.DOCKER_USERNAME }}/ci-cd-clean-app:latest

  cd:
    name: CD - Deploy to EC2
    runs-on: ubuntu-latest
    needs: ci

    steps:
      - name: Deploy application to EC2
        uses: appleboy/ssh-action@v1.0.3        # appleboy/ssh-action is used to securely SSH into a remote EC2 instance from GitHub Actions using an SSH private key.
        with:
          host: ${{ secrets.EC2_HOST }}
          username: ${{ secrets.EC2_USER }}
          key: ${{ secrets.EC2_KEY }}
          script: |                                # uses → calling an action

                                                     script → input parameter of that action
                                                      
                                                     Keyword	   Where commands run

						      run	   GitHub Actions runner

						      script	   Remote server (EC2)


            docker stop ci-cd-clean-app || true    # Stop the container if it is running.

            docker rm ci-cd-clean-app || true      # Remove the conatiner if it exist

            docker pull ${{ secrets.DOCKER_USERNAME }}/ci-cd-clean-app:latest   # pull the docker image
                                                                                 
                                                                                Stop old container-> Remove old container-> Pull new image-> Run new container

            docker run -d \                                             # Create and start a new container from an image (start the container, not image)
              --name ci-cd-clean-app \
              -p 5000:5000 \                                            # port mapping,-p 5000:5000 exposes the container’s port 5000 on the host machine, allowing external access to the          									  application
              ${{ secrets.DOCKER_USERNAME }}/ci-cd-clean-app:latest



| Command      | Removes container? | Removes image? |
| ------------ | ------------------ | -------------- |
| `docker rm`  | ✅ Yes              | ❌ No           |
| `docker rmi` | ❌ No               | ✅ Yes          |


| Part           | Meaning          |
| -------------- | ---------------- |
| `-p`           | Publish port     |
| `5000` (left)  | Host (EC2) port  |
| `5000` (right) | Container port   |
| `\`            | Continue command |
